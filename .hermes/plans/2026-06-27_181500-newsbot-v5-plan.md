#### Task 6: Consertar deduplicacao (substring → comparacao exata + fuzzy)

**Objetivo:** Eliminar falsos positivos onde "Tom Hardy" bloqueia "Tom Hardy Is Crossing Over..."

**Files:**
- Modify: `main.py:53-59`

**Step 1: Reescrever is_duplicate()**

```python
def is_duplicate(title, history):
    """Dedup por similaridade de titulo (hash dos primeiros 100 chars)."""
    clean = clean_html(title).strip().lower()[:100]
    for h in history:
        h_title = h if isinstance(h, str) else h.get('title', '')
        h_clean = clean_html(h_title).strip().lower()[:100]
        # So considera duplicata se for EXATAMENTE igual
        # ou se um contem o outro COMPLETO (nao substring parcial)
        if clean == h_clean:
            return True
        # Fuzzy: se diferenca de tamanho < 20% e similaridade > 85%
        len_diff = abs(len(clean) - len(h_clean)) / max(len(clean), len(h_clean), 1)
        if len_diff < 0.2:
            # Conta palavras em comum
            words_a = set(clean.split())
            words_b = set(h_clean.split())
            if words_a and words_b:
                overlap = len(words_a & words_b) / min(len(words_a), len(words_b))
                if overlap > 0.85:
                    return True
    return False
```

```python
def summarize_light(text, max_chars=300):
    """Resumo leve: primeiras 2-3 frases do texto原始."""
    if not text:
        return ""
    clean = clean_html(text)
    if len(clean) <= max_chars:
        return clean
    # Pega ate o 3o ponto final
    sentences = re.split(r'(?<=[.!?])\s+', clean)
    result = ""
    for s in sentences[:3]:
        if len(result) + len(s) <= max_chars:
            result += s + " "
        else:
            break
    return result.strip()
```

```python
def send_telegram_long_message(message, max_chunk=3800):
    """Envia mensagem longa em chunks numerados."""
    if len(message) <= max_chunk:
        return send_telegram_message(message)
    
    # Split por paragrafos para nao quebrar no meio de linha
    paragraphs = message.split('\n\n')
    chunks = []
    current = ""
    for p in paragraphs:
        if len(current) + len(p) + 2 <= max_chunk:
            current += p + "\n\n" if current else p
        else:
            if current:
                chunks.append(current.strip())
            current = p
    
    if current:
        chunks.append(current.strip())
    
    total = len(chunks)
    success = True
    for i, chunk in enumerate(chunks, 1):
        prefix = f"({i}/{total}) " if total > 1 else ""
        if not send_telegram_message(prefix + chunk):
            success = False
    
    return success
```
